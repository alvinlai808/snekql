Introduction to SnekQL

SnekQL is a multi-paradigm query language oriented towards data science. Pragmatic and intuitive, our language is intended to provide a simple yet powerful way of manipulating (representing and visualizing) data.

fnc fibonacci(n):
    if n <= 1:
        return n
    else:
        return(fibonacci(n-1) + fibonacci(n-2))

fnc addValuesToColumn():
    for i in range(numberOfTerms):

Table Creation for Pythagorean Triples:

legAValues = []
legBValues = []
hypotenuseValues = []

legsA = column("Legs", legAValues, True, True)
legsB = column("Legs", legbValues, True, True)
hypotenuses = column(
    "Hyptonuses",
    hypotenuseValues,
    True,
    True
)

pythagoreanTriples = tabulate(legsA, legsB, hypotenuses)

fnc pythagoreanTriplets(limit):
    hypotenuse = 0
    two = 2

    while hypotenuse < limit:
        for i in range(1, two) :
            legA = two * two - i * i
            legB = 2 * two * i
            hypotenuse = two * two + i * i

            if hypotenuse > limit:
                break
            legsA.addRow(legA)
            legsB.addRow(legB)
            hypotenuses.addRow(hypotenuse)
        two += 1

pythagoreanTriplets(20)

Sorting Column with Bubble Sort Algorithm:

values = column(
    "List of Numbers",
    [22, 23434, 2332, 7543, 10132], 
    True,
    True
)

fnc bubbleSort(column):
    length  = 0
    switch  = true

    for i in column:
        length += 1

    while switch:
        switch = false
        for i in range(length - 1):
            if column[i] > column[i + 1]:
                temp = column[i]
                column[i] = column[i + 1]
                column[i + 1] = temp
                switch  = true

bubbleSort(values)
